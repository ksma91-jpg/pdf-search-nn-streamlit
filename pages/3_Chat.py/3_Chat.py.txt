import streamlit as st

st.title("💬 Chat basado en evidencia")

st.info(
    "Como no tenemos API del LLM interno, esta sección preparará contexto "
    "relevante para copiar y pegar en el chat corporativo aprobado."
)

question = st.text_area("Pregunta")

if st.button("Preparar contexto"):
    if not question.strip():
        st.warning("Escribe una pregunta.")
    else:
        st.info("En los siguientes pasos conectaremos esta sección a la búsqueda semántica.")
        st.write(f"Pregunta: {question}")
